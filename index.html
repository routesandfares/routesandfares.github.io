# Search a flight from London Heathrow to New York JFK on January, the 1st
curl -X POST \
    -H 'Content-Type: application/json' \
    -d '{
          "flights": [
            { 
              "from": "LHR",
              "to"  : "JFK",
              "departure" : "2016-01-01",
              "passengers": [{ "category": "ADT "}]
            }
          ]
        }' \
